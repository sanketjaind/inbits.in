Wscript For Auto Email Sign Up

   Set IE = CreateObject("InternetExplorer.Application")
   IE.Visible = 1 
   IE.navigate "https://mail.tcs.com"
   Do While (IE.Busy)
    WScript.Sleep 10
   Loop
   Set Helem = IE.document.getElementByID("username")
   Helem.Value = "1216537" ' change this to yours
   Set Helem = IE.document.getElementByID("password")
   Helem.Value = "" ' change this to yours
   Set Helem = IE.document.Forms(0)
   Helem.Submit

Wscript For Wake Up Screen For Certain Time

   Set ws = CreateObject("WScript.Shell")
startTime = Now
'H = Hours
'N = Minutes
'S = Seconds
endTime = DateAdd("N",90,startTime)

Do Until Now >= endTime
    Wscript.Sleep 180000
    ws.SendKeys "{F15}"
Loop
